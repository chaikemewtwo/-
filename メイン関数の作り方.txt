	DebugStage ds;

	// メッセージを受け取る
	while (ProcessMessage() == true) {

		// 標準エスケープキー
		if (GetKeyState(VK_ESCAPE) & 0x8000) {
			break;
		}

		// 描画開始
		if (DrawStart() == true) {
			
		   // ここに処理を書く

		}

		// 描画の終わり
		DrawEnd();
	}

	// 解放処理は必ず行う
	Texture::Release();
	GraphicsRelease();

	return 0;
}
